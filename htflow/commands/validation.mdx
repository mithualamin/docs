## Validation Configuration

Customize validation rules in your config:

```javascript htflow.config.js
module.exports = {
  validation: {
    // Enable/disable validators
    html: true,
    css: true,
    javascript: true,
    
    // Strictness level
    strict: false,  // Set to true for production
    
    // Custom rules
    rules: {
      // HTML rules
      html: {
        requireAltText: true,
        semanticElements: true,
        accessibility: "warn"  // "error", "warn", or "off"
      },
      
      // CSS rules
      css: {
        vendorPrefixes: "auto",
        browserSupport: ["last 2 versions"],
        webflowClasses: true
      },
      
      // JavaScript rules
      javascript: {
        es6: true,
        strict: false,
        browserCompat: "modern"
      }
    },
    
    // Ignore patterns
    ignore: [
      "**/node_modules/**",
      "**/vendor/**",
      "**/*.min.*"
    ]
  }
};
```

## Audit Configuration

Configure audit behavior:

```javascript htflow.config.js
module.exports = {
  audit: {
    // Output settings
    outputFormat: "html",  // "html", "json", or "markdown"
    outputPath: "./audit-report.html",
    
    // Thresholds
    thresholds: {
      performance: 80,
      accessibility: 90,
      bestPractices: 85,
      seo: 80,
      webflow: 100
    },
    
    // What to audit
    checks: {
      performance: true,
      accessibility: true,
      bestPractices: true,
      seo: true,
      webflow: true
    },
    
    // Fail on threshold
    failOnThreshold: false
  }
};
```

## Fix Common Issues

<AccordionGroup>
  <Accordion title="Missing Alt Text" icon="image">
    **Problem:** Images without alt attributes
    
    **Fix:**
    ```html
    <!-- ‚ùå Before -->
    <img src="image.jpg">
    
    <!-- ‚úÖ After -->
    <img src="image.jpg" alt="Descriptive text">
    ```
  </Accordion>
  
  <Accordion title="CSS Vendor Prefixes" icon="code">
    **Problem:** Missing vendor prefixes
    
    **Fix:**
    ```css
    /* ‚ùå Before */
    .element {
        backdrop-filter: blur(10px);
    }
    
    /* ‚úÖ After */
    .element {
        -webkit-backdrop-filter: blur(10px);
        backdrop-filter: blur(10px);
    }
    ```
  </Accordion>
  
  <Accordion title="JavaScript Compatibility" icon="js">
    **Problem:** Modern JavaScript not supported
    
    **Fix:**
    ```javascript
    // ‚ùå Before (ES6)
    const data = array.map(item => item.value);
    
    // ‚úÖ After (ES5)
    var data = array.map(function(item) {
        return item.value;
    });
    ```
  </Accordion>
  
  <Accordion title="Improper Heading Hierarchy" icon="heading">
    **Problem:** Skipping heading levels
    
    **Fix:**
    ```html
    <!-- ‚ùå Before -->
    <h1>Title</h1>
    <h3>Subtitle</h3>
    
    <!-- ‚úÖ After -->
    <h1>Title</h1>
    <h2>Subtitle</h2>
    ```
  </Accordion>
</AccordionGroup>

## Continuous Integration

Integrate validation in your CI/CD:

```yaml .github/workflows/validate.yml
name: HTFlow Validation
on: [push, pull_request]

jobs:
  validate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - uses: actions/setup-node@v2
      - run: npm install -g htflow-cli
      - run: htflow validate
      - run: htflow audit
```
